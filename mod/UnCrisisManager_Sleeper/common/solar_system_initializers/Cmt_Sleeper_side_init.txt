	init_effect = {
		create_fe_home_citadel = yes

				count_fleet_in_system = {

					limit = {
						is_civilian = no

					}
				}
			}
			spawn_natural_wormhole = {
				bypass_type = wormhole
				random_pos = yes
				orbit_angle = 360
			}
			prev = {
				spawn_natural_wormhole = {
					bypass_type = wormhole
					random_pos = yes
				}
				link_wormholes = prev
			}
		}
	}
