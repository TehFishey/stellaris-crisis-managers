		trigger = { event_target:CmtGlobalVar = { check_variable = { which = CmtVarSleeperMultiple value = 3 } } }
	}

		text = CrisisManagerMenu.4510.desc4
		trigger = { event_target:CmtGlobalVar = { check_variable = { which = CmtVarSleeperMultiple value = 4 } } }
	}
	desc = {
		text = CrisisManagerMenu.4510.desc5
		trigger = { event_target:CmtGlobalVar = { check_variable = { which = CmtVarSleeperMultiple value = 5 } } }
	}
	desc = {
		text = CrisisManagerMenu.4510.desc
		trigger = {
			event_target:CmtGlobalVar = {
				OR = {
					check_variable = { which = CmtVarSleeperMultiple value < 1 }
					check_variable = { which = CmtVarSleeperMultiple value > 5 }
				}
			}
		}
	}
	picture = GFX_evt_fallen_empire
	is_triggered_only = yes

	after = {
		hidden_effect = {
			if = {
				limit = { has_country_flag = CmtFlagGoBack }
				remove_country_flag = CmtFlagGoBack
				country_event = { id = CrisisManagerMenu.4000 }
			} else_if = {
				limit = { has_country_flag = CmtFlagIndividual }
				country_event = { id = CrisisManagerMenu.1024 }
			} else_if = {
				limit = { event_target:CmtGlobalVar = { check_variable = { which = CmtVarSleeperMultiple value = 1 } } }
				country_event = { id = CrisisManagerMenu.4000 }
			} else_if = {
				limit = {
					event_target:CmtGlobalVar = {
						OR = {
							check_variable = { which = CmtVarSleeperMultiple value = 4 }
							check_variable = { which = CmtVarSleeperMultiple value = 5 }
						}
					}
				}
				country_event = { id = CrisisManagerMenu.4520 }
			} else = { country_event = { id = CrisisManagerMenu.4530 } }
		}
	}

	option = {
		name = CmtTextKeepCurrent.name
		custom_tooltip = CmtTextKeepCurrent.tooltip
	}

		custom_tooltip = CrisisManagerMenu.4510a.tooltip
