### Check Progress Speed of Opening Event Chain
country_event = {
	id = CrisisManagerControl.3000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_flag = ai_invasion_happened }
	}
	
	immediate = {
		endgame_telemetry = ai_revolt
		set_global_flag = galactic_crisis_happened
		set_global_flag = ai_invasion_happened
		set_global_flag = ai_ghost_signal
		set_global_flag = CmtFlagContingencyTriggerd
		observer_event = { id = observer.41 }
		
		event_target:CmtGlobalVar = {
			if = {
				limit = { check_variable = { which = CmtVarOpeningEventSpeed value = 1 } }
				country_event = { id = CrisisManagerControl.3010 }
			} else_if = {
				limit = { check_variable = { which = CmtVarOpeningEventSpeed value = 2 } }
				country_event = { id = CrisisManagerControl.3020 }
			} else_if = {
				limit = { check_variable = { which = CmtVarOpeningEventSpeed value = 4 } }
				country_event = { id = CrisisManagerControl.3040 }
			} else_if = {
				limit = { check_variable = { which = CmtVarOpeningEventSpeed value = 5 } }
				country_event = { id = CrisisManagerControl.3050 }
			} else_if = {
				limit = { check_variable = { which = CmtVarOpeningEventSpeed value = 6 } }
				country_event = { id = CrisisManagerControl.3060 }
			} else = { country_event = { id = CrisisManagerControl.3030 } }
		}
	}
}
